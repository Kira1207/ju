import java.util.Scanner;

public class Main {

	public static void main(String[] args) {

		Scanner sc = new Scanner(System.in);
		do {
			System.out.println("Introduce un nÃºmero");
			n = sc.nextInt();
			total++;
			if(n>mayor) {
				mayor = n;

		System.out.println("El total de num fallados es "+ fallos);
	}

}
